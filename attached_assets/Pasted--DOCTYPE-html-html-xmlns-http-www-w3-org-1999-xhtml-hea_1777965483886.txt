</p>

  <li>b.	Select Country (option available only for CC User. For Individual /TA’s this option is disabled.)</li>

  <li>i.	Appointment confirmation: Click on “Confirm” button in the final calendar page</li>
  <li>j.	An email & SMS will be triggered to the applicant/s after appointment is confirmed. </li>
</ul>

</p>

  <li>b.	User can select the applicant/s to reschedule and click on continue to proceed to final calendar to select date & time</li>
  <li>c.	Upon selection date & time, can confirm the appointment by clicking confirm button.</li>
  <li>d.	Appointment confirmation message will be displayed</li>
  <li>e.	An email & SMS will be triggered to the applicant/s</li>
</ul>
<p>

</p>

  <li>b.	User can select the applicant/s to cancel and click on continue.</li>
  <li>c.	Appointment cancellation confirmation message will be displayed</li>
  <li>d.	An email & SMS will be triggered to the applicant/s</li>
</ul>
<p>
  <b>4.	Print or Email Appointment Letter:</b>  To print an appointment letter, user need to have the reference number with valid future appointment. User can click on “Print an appointment” link.
</p>
<ul>
  <li>a.	Enter the reference number and click on submit to get the applicant/s details</li>
  <li>b.	User can select the applicant/s to print and click on “Download appointment letter” button.</li>
  <li>c.	An email & SMS will be triggered to the applicant</li>

</ul>
<p>
  <b>5.	Retrieve Incomplete Appointment:</b>  To retrieve an incomplete appointment, user need to have the reference number. User can click on “Retrieve Incomplete appointment” link.
</p>

    function SetCountDown(data) {
        if (navigator.userAgent.toLowerCase().indexOf('firefox') > -1) {

            var str = data.DownTimeDateTime;
            data.DownTimeDateTime = str.split(".").join("/");
        }
        $(".timezone p:nth-of-type(1)").html('The Next System Maintenance Details.');
        $(".timezone p:nth-of-type(2)").html(String.format('Start Time: {0} GMT', data.DownDateTime));
        $(".timezone p:nth-of-type(3)").html(String.format('Est. End Time: {0} GMT', data.UpDateTime));
        $("#countdown").html('Count Down' + ' | ');
        var note = $('#note'),
            ts = new Date(Date.parse(data.DownTimeDateTime)),
            newYear = true;
        
        if ((new Date()) > ts) {
            // The new year is here! Count towards something else.
            // Notice the *1000 at the end - time must be in milliseconds
            ts = (new Date()).getTime() + 10 * 24 * 60 * 60 * 1000;

            newYear = false;
        }

        $('#countdown').countdown({
            timestamp: ts,
            callback: function (days, hours, minutes, seconds) {
                var message = "";
                message += days + ' day(s)' + ", ";
                message += hours + ' hour(s)' + ", ";
                message += minutes + ' minute(s)' + ' and ';
                message += seconds + ' second(s)' + " <br />";
                if (days == 0 & hours == 0 & minutes == 0 & seconds == 0) {
                    $(".timezone").css("display", "none");
                }
                note.html(message);
            }
        });
    }

    String.format = function () {
        // The string containing the format items (e.g. "{0}")
        // will and always has to be the first argument.
        var theString = arguments[0];

        // start with the second argument (i = 1)
        for (var i = 1; i < arguments.length; i++) {
            // "gm" = RegEx options for Global search (more than one instance)
            // and for Multiline search
            var regEx = new RegExp("\\{" + (i - 1) + "\\}", "gm");
            theString = theString.replace(regEx, arguments[i]);
        }

        return theString;
    }

            (function ($) {
                'use strict';
                var Globalconfig = {
                    seperator: '/'
                };
                var Applicantconfig = {
                    seperator: '-'
                };

                $.fn.formatToDate = function (config) {
                    var $this = this;

                    $.each($this, function () {

                        var $$this = $(this),
                            cnfg = config || Globalconfig;

                        $$this.on('keydown', function (evt) {

                            var value = $$this.val(),
                                splitArry = [];

                            splitArry = value.split("");

                            if (!(splitArry.length < 10 && isNumeric(evt) || isDeleteOp(evt)) && !isTabOp(evt)) {
                                evt.preventDefault();
                            }
                        });

                        $$this.on('keyup', function (e) {

                            var textSoFar = $(this).val();
                            if (e.keyCode != 191) {
                                if (e.keyCode != 8) {
                                    if (textSoFar.length == 2 || textSoFar.length == 5) {
                                        $(this).val(textSoFar + "/");
                                    }
                                    //to handle copy & paste of 8 digit
                                    else if (e.keyCode == 86 && textSoFar.length == 8) {
                                        $(this).val(textSoFar.substr(0, 2) + "/" + textSoFar.substr(2, 2) + "/" + textSoFar.substr(4, 4));
                                    }
                                } else {
                                    //backspace would skip the slashes and just remove the numbers
                                    if (textSoFar.length == 5) {
                                        $(this).val(textSoFar.substring(0, 4));
                                    } else if (textSoFar.length == 2) {
                                        $(this).val(textSoFar.substring(0, 1));
                                    }
                                }
                            } else {
                                //remove slashes to avoid 12//01/2014
                                $(this).val(textSoFar.substring(0, textSoFar.length - 1));
                            }

                        });

                        return $this;


                        //private functions

                        function isNumeric(e) {
                            return (e.keyCode >= 48 && e.keyCode <= 57) || (e.keyCode >= 96 && e.keyCode <= 105);
                        }

                        function isDeleteOp(e) {
                            return (e.keyCode === 46 || e.keyCode === 8);
                        }

                        function isTabOp(e) {
                            return e.keyCode === 9 || e.keyCode === 191 || e.keyCode === 37 || e.keyCode === 39;
                        }
                    });

                };
